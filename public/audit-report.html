0. Modelo de Dominio OST

Project (contexto organizacional — multi-OST)
  +-- Oportunidad (problema / necesidad del usuario)
        |-- Evidencia             -> citas, hechos u observaciones que la sustentan
        |-- Solucion              -> idea de producto/feature para resolver la oportunidad
        |     +-- Supuesto        -> creencia falseable (deseabilidad/viabilidad/factibilidad/usabilidad)
        |           +-- Experimento -> prueba disenada (type, success_criterion, effort, impact)
        +-- Evaluacion IA         -> analisis Claude + conversacion de refinamiento
              +-- Priorizacion    -> top-3 experimentos (menor esfuerzo, mayor impacto)
EntidadPadreSemantica
opportunitiesprojectsUna oportunidad pertenece a un proyecto
opportunity_evidenceopportunitiesJustifica la oportunidad con datos reales
solutionsopportunitiesIdea de solucion para abordar la oportunidad
assumptionssolutionsCreencia falseable categorizada por dimension
experimentsassumptionsPrueba disenada con criterio de exito, esfuerzo e impacto
ai_evaluationsopportunitiesEvaluacion Claude — audit trail inmutable
ai_conversation_messagesai_evaluationsHilo de refinamiento post-evaluacion con Claude
project_membersprojects + auth.usersRol del usuario en el proyecto

1. Schema de Persistencia (SQL)

CREATE TABLE projects (
  id UUID PK, name TEXT NOT NULL, description TEXT,
  created_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT now(), updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE project_members (
  project_id UUID FK, user_id UUID FK,
  role TEXT CHECK (role IN ('admin', 'usuario', 'viewer')),
  invited_at TIMESTAMPTZ DEFAULT now(), PRIMARY KEY (project_id, user_id)
);

CREATE TABLE business_context (
  id UUID PK, project_id UUID FK, content TEXT NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE opportunities (
  id UUID PK, project_id UUID FK, name TEXT NOT NULL,
  description TEXT, outcome TEXT, archived BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ, updated_at TIMESTAMPTZ
);

CREATE TABLE opportunity_evidence (
  id UUID PK, opportunity_id UUID FK,
  type TEXT CHECK (type IN ('cita','hecho','observacion')),
  content TEXT NOT NULL, source TEXT, -- obligatorio si type='cita'
  created_at TIMESTAMPTZ
);

CREATE TABLE solutions (
  id UUID PK, opportunity_id UUID FK, name TEXT NOT NULL,
  description TEXT,
  created_at TIMESTAMPTZ, updated_at TIMESTAMPTZ
);

CREATE TABLE assumptions (
  id UUID PK, solution_id UUID FK, description TEXT NOT NULL,
  category TEXT CHECK (category IN ('deseabilidad','viabilidad','factibilidad','usabilidad')),
  status TEXT CHECK (status IN ('pendiente','validado','invalidado')),
  result TEXT, created_at TIMESTAMPTZ, updated_at TIMESTAMPTZ
);

CREATE TABLE experiments (
  id UUID PK, assumption_id UUID FK,
  type TEXT CHECK (type IN ('entrevista','prototipo','smoke_test','prueba_usabilidad','otro')),
  description TEXT NOT NULL, success_criterion TEXT NOT NULL,
  effort TEXT CHECK (effort IN ('bajo','medio','alto')),
  impact TEXT CHECK (impact IN ('bajo','medio','alto')),
  status TEXT CHECK (status IN ('to do','en curso','terminada')),
  result TEXT, created_at TIMESTAMPTZ, updated_at TIMESTAMPTZ
);

CREATE TABLE ai_evaluations (
  id UUID PK, opportunity_id UUID FK,
  prompt_snapshot TEXT NOT NULL, evaluation_text TEXT NOT NULL,
  created_at TIMESTAMPTZ
);

CREATE TABLE ai_conversation_messages (
  id UUID PK, evaluation_id UUID FK,
  role TEXT CHECK (role IN ('user','assistant')),
  content TEXT NOT NULL, created_at TIMESTAMPTZ
);

-- RLS: todas las tablas con ROW LEVEL SECURITY habilitado
-- Helper: user_role_in_project(p_id) -> retorna el rol del usuario
-- Policy: members can read, editors (admin/usuario) can write

REQ-003State-drivenMIENTRAS rol=viewer -> deshabilitar todos los controles de escritura.
REQ-004State-drivenMIENTRAS rol=admin -> habilitar gestion de miembros.
REQ-005UnwantedSI viewer intenta escribir -> rechazar con 403.
REQ-006Event-drivenCUANDO admin invite miembro -> insertar en project_members y notificar.
